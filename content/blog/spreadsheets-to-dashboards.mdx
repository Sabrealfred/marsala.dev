WITH first_purchase AS (
  SELECT
    customer_email,

  FROM raw_orders
  WHERE status = 'completed'
  GROUP BY customer_email
),
cohort_revenue AS (
  SELECT
    fp.cohort_month,
    DATE_TRUNC('month', o.created_at) as revenue_month,

    SUM(o.total_amount) as revenue
  FROM raw_orders o

  WHERE o.status = 'completed'

)

**Transformaci√≥n 2: Marketing Attribution**

```sql
-- ROI por canal de marketing
WITH ad_spend AS (
  SELECT
    DATE_TRUNC('day', date) as day,
    platform,
    SUM(spend) as total_spend
  FROM raw_ad_spend
  GROUP BY day, platform
),
attributed_sales AS (
  SELECT
    DATE_TRUNC('day', created_at) as day,
    source as platform,
    SUM(total_amount) as revenue,
    COUNT(*) as orders
  FROM raw_orders
  WHERE status = 'completed'
  GROUP BY day, platform
)
SELECT
  s.day,
  s.platform,
  s.total_spend,
  COALESCE(a.revenue, 0) as revenue,
  COALESCE(a.orders, 0) as orders,
  CASE
    WHEN s.total_spend > 0
    THEN ROUND((COALESCE(a.revenue, 0) / s.total_spend - 1) * 100, 2)
    ELSE NULL
  END as roas_percentage
FROM ad_spend s
LEFT JOIN attributed_sales a
  ON s.day = a.day AND s.platform = a.platform
ORDER BY s.day DESC;
```
